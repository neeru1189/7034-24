15. Program to sort the elements of an array in descending order


        #include<iostream>
        using namespace std;
        
        int main() {
            int arr[20],t,n,i;
            cout<<"Enter number of elements (1-20) : "<<endl;
            cin>>n;
            for(i=0;i<n;i++) {
                cout<<"Enter element "<<i+1<<" : ";
                cin>>arr[i];
            }
            for(i=0;i<n;i++) {
                for(int j=i+1;j<n;j++) {
                    if(arr[i]<arr[j]) {
                        t=arr[i];
                        arr[i]=arr[j];
                        arr[j]=t;
                    }
                }
            }
            cout<<endl<<"Array in descending order : "<<endl;
            for(i=0;i<n;i++) {
                cout<<"\t"<<arr[i];
            }
        }
