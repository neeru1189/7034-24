12. Program to print the sum of all the items of the array


        #include<iostream>
        using namespace std;
        
        int main() {
            int arr[20],n,i,sum=0;
            cout<<"Enter number of elements (1-20) : "<<endl;
            cin>>n;
            for(i=0;i<n;i++) {
                cout<<"Enter element "<<i+1<<" : ";
                cin>>arr[i];
                sum=sum+arr[i];
            }
            cout<<endl<<"Sum of elements in array : "<<sum;
        }
