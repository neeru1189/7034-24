5. Program to print the elements of an array 


        #include<iostream>
        using namespace std;
        
        int main() {
            int arr[5],i;
            cout<<"Enter 5 elements of array : "<<endl;
            for(i=0;i<5;i++) {
                cout<<"Enter element "<<i+1<<" : ";
                cin>>arr[i];
            }
            cout<<endl<<"Printing array elements : ";
            for(i=0;i<5;i++) {
                cout<<endl<<"Element "<<i+1<<" : "<<arr[i];
            }
        }
