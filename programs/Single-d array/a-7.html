7. Program to print the elements of an array present on even position


        #include<iostream>
        using namespace std;
        
        int main() {
            int arr[5],i;
            cout<<"Enter 5 elements of array : "<<endl;
            for(i=0;i<5;i++) {
                cout<<"Enter element "<<i+1<<" : ";
                cin>>arr[i];
            }
            cout<<endl<<"Printing elements at even position : "<<endl;
            for(i=0;i<5;i++) {
                if(i%2!=0) {
                    cout<<"\t"<<arr[i];
                }
            }
        }

Output
