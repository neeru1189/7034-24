6. Program to print the elements of an array in reverse order


        #include<iostream>
        using namespace std;
        
        int main() {
            int arr[5],i;
            cout<<"Enter 5 elements of array : "<<endl;
            for(i=0;i<5;i++) {
                cout<<"Enter element "<<i+1<<" : ";
                cin>>arr[i];
            }
            cout<<endl<<"Printing array elements in reverse order: "<<endl;
            for(i=4;i>=0;i--) {
                cout<<" \t "<<arr[i];
            }
        }

Output

        Enter element 3 : 30

        Enter element 5 : 50
