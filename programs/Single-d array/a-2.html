2. Program to find the frequency of each element in the array


        #include<iostream>
        using namespace std;
        
        int main() {
            int arr[5],i,j,c=0;
            cout<<"Enter 5 elements of array : "<<endl;
            for(i=0;i<5;i++) {
                cout<<"Enter element "<<i+1<<" : ";
                cin>>arr[i];
            }
            cout<<endl<<"Frequency of numbers : ";
            for(i=0;i<5;i++) {
                j = 0;
                c = 0;
                while(j<5) {
                    if(arr[i]==arr[j]) {
                        c++;
                    }
                    j++;
                }
                cout<<endl<<"Frequency of "<<arr[i]<<" : "<<c;
            }
        }
