8. C++ Program to convert double to String 


        #include<iostream>
        #include<string>
        #include<sstream>
        using namespace std;
        int main(){
            double d;
            stringstream s1;
            string st="";
            cout<<"Enter the double to convert it into string : ";
            cin>>d;
            s1<<d;
            s1>>st;
            cout<<"String after conversion to double : "<<st;
            return 0;
        }

Output


            Enter the double to convert it into string : 3.14
            String after conversion to double : 3.14
