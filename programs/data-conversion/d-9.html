9. C++ Program to convert char to int 


        #include<iostream>
        #include<string>
        #include<sstream>
        using namespace std;
        int main(){
            int n;
            char ch;
            cout<<"Enter the integer character : ";
            cin>>ch;
            n=ch-'0';
            cout<<"Character after conversion into int : "<<n;
            return 0;
        }

Output


            Enter the integer character : 5
            Character after conversion into int : 5
