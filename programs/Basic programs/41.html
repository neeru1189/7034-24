41. C++ program to sum of N numbers


        #include<iostream>
        using namespace std;
        
        int main() {
            int n;
            cout<<"Enter any number : "<<endl;
            cin>>n;
            int sum=0;
            for(int i=0;i<=n;i++) {
                sum=sum+i;
            }
            cout<<endl<<"Sum of "<<n<<" natural numbers : "<<sum;
        }

Output


        Enter any number : 
        5
        Sum of 5 natural numbers : 15
