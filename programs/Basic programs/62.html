62. Write a C++ program that reads an positive integer and count the number of digits 


        #include<iostream>
        using namespace std;
        
        int main()
        {
            int n, c = 0;
            int m;
            cout << "Enter any positive number : ";
            cin >> n;
            if (n > 0)
            {
                m = n;
                while (m != 0)
                {
                    m = m / 10;
                    c++;
                }
                cout << endl << "Number of digits in " << n << " are : " << c;
            }
            else
            {
                cout << endl << "Entered number is negative";
            }
        }

Output


        Enter any positive number : -123
        Entered number is negative
