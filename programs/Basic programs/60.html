60. Write a C++ program to display the multiplication table of a given integer 


        #include<iostream>
        using namespace std;
        
        int main()
        {
            int n;
            cout << "Enter any number : ";
            cin >> n;
            cout << endl << "Multiplication table of " << n << " : ";
            for (int i = 1; i <= 10; i++)
            {
                cout << endl << n << " X " << i << " = " << (n * i);
            }
        }
