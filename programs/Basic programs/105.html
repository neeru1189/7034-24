105. Write a C++ method to find GCD and LCM of Two Numbers 

        
          #include<iostream>
          using namespace std;
          int main(){
            int a,b,r;
            cout<<"Enter the number A and B : ";
            cin>>a>>b;
            if(a>b){
                r=b;
            }
            else{
                r=a;
            }
            while(r>0){
                if(a%r==0 && b%r==0){
                    break;
                }
                r--;
            }
            cout<<"GCD of "<<a<<" and "<<b<<" : "<<r<<endl;
            cout<<"LCM of "<<a<<" and "<<b<<" : "<<(a*b)/r;
            return 0;
          }

Output

            
              GCD of 12 and 18 : 6
              LCM of 12 and 18 : 36
