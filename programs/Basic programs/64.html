64. Write a C++ program that accepts three numbers from the user and check if numbers are in "increasing” or “decreasing” order. 


        #include<iostream>
        using namespace std;
        
        int main()
        {
            int x, y, z;
            cout << "Enter first number : ";
            cin >> x;
            cout << "Enter second number : ";
            cin >> y;
            cout << "Enter third number : ";
            cin >> z;
            if (x > y && y > z)
            {
                cout << endl << "Numbers are in Decreasing Order";
            }
            else if (x < y && y < z)
            {
                cout << endl << "Numbers are in Increasing Order";
            }
            else
            {
                cout << endl << "Numbers are in Random Order";
            }
        }

Output


        Enter first number : 3
        Enter second number : 5
        Enter third number : 2
        Numbers are in Random Order
