        #include<iostream>
        using namespace std;
        
        int main()
        {
            int d;
            cout << "Enter week number (1-7) : ";
            cin >> d;
            switch(d)
            {
                case 1:
                    cout << endl << "Monday";
                    break;
                case 2:
                    cout << endl << "Tuesday";
                    break;
                case 3:
                    cout << endl << "Wednesday";
                    break;
                case 4:
                    cout << endl << "Thursday";
                    break;
                case 5:
                    cout << endl << "Friday";
                    break;
                case 6:
                    cout << endl << "Saturday";
                    break;
                case 7:
                    cout << endl << "Sunday";
                    break;
                default:
                    cout << endl << "Invalid week number!";
            }
        }
