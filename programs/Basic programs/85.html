85. Write a C++ program that prompts the user to input an integer and then outputs the number with the digits reversed order 

        
        #include<iostream>
        using namespace std;
        
        int main()
        {
            int n, rem, rev = 0;
            cout << "Enter any integer : ";
            cin >> n;
            int m = n;
            while (m != 0)
            {
                rem = m % 10;
                rev = rev * 10 + rem;
                m = m / 10;
            }
            cout << endl << "Number with reverse digits : " << rev;
        }

Output

            
            Enter any integer: 12345
            
            Number with reverse digits: 54321
