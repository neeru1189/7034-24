43. C++ Program to calculate electricity bill


        #include<iostream>
        using namespace std;
        
        int main() {
            int p,uc,ou,nu,puc;
            float tax,bill;
            cout<<"Enter Old Unit : "<<endl;
            cin>>ou;
            cout<<"Enter New Unit : "<<endl;
            cin>>nu;
            cout<<"Enter Per Unit Charge : "<<endl;
            cin>>puc;
            uc=nu-ou;
            p=uc*puc;
            tax=(float)((p*12)/100);
            bill=p+tax;
            cout<<endl<<"Total Bill : "<<bill;
        }
