83. Write a C++ program to print multiplication table of given number 

        
        #include<iostream>
        using namespace std;
        
        int main()
        {
            int n;
            cout << "Enter any number : ";
            cin >> n;
            cout << endl << "Multiplication table of " << n << " : ";
            for (int i = 1; i <= 10; i++)
            {
                cout << endl << n << " X " << i << " = " << (n * i);
            }
        }
