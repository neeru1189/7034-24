112. Write a C++ program to convert a float value to absolute value

        
          #include<iostream>
          using namespace std;
          int main(){
              float f;
              cout<<"Enter the float value : ";
              cin>>f;
              if(f>=0){
                  cout<<"Absolute value : "<<f;
              }
              else{
                  cout<<"Absolute value : "<<-f;
              }
              return 0;
          }

Output

        
          Enter the float value: -3.14
          Absolute value: 3.14
