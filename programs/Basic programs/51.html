51. Write a C++ program to find the number of days in a month


        #include<iostream>
        using namespace std;
        
        int main()
        {
            int n;
            cout << "Enter month number : ";
            cin >> n;
            switch(n)
            {
                case 1:
                case 3:
                case 5:
                case 7:
                case 8:
                case 10:
                case 12:
                    cout << "There are 31 days";
                    break;
                case 2:
                    cout << "There are 28/29 days";
                    break;
                case 4:
                case 6:
                case 9:
                case 11:
                    cout << "There are 30 days";
                    break;
                default:
                    cout << "Invalid Month Number";
            }
        }

Output


        Enter month number : 7
        There are 31 days
