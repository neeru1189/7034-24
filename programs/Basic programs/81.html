81. Write a C++ program to print numbers from 1 to 10 using loop 

        
        #include<iostream>
        using namespace std;
        
        int main()
        {
            int i;
            cout << "Number between 1-10 : ";
            for (i = 1; i <= 10; i++)
            {
                cout << endl << i;
            }
        }
