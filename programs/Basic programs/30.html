30. C++ Program to find area of parallelogram


        #include<iostream>
        using namespace std;
        
        int main() {
            int b,h;
            cout<<"Enter Base of Parallelogram : "<<endl;
            cin>>b;
            cout<<"Enter Height of Parallelogram : "<<endl;
            cin>>h;
            int area=b*h;
            cout<<endl<<"Area of Parallelogram : "<<area<<" square units";
        }
