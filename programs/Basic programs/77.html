77. Write a C++ program to reverse a String 


        #include<iostream>
        #include<string.h>
        using namespace std;
        
        int main() {
            char str[20];
            int l, i;
            cout << "Enter a string : ";
            cin >> str;
            l = strlen(str);
            cout << endl << "String after reversing : ";
            for (i = l; i >= 0; i--) {
                cout << str[i];
            }
        }

Output


        Enter a string: hello
        String after reversing: olleh
