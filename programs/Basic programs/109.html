109. Write a C++ program to test if a double number is an integer

        
          #include<iostream>
          using namespace std;
          int main(){
              int n;
              float f;
              cout<<"Enter the Double number : "; 
              cin>>f;
              n=f;
              if(n==f){
                  cout<<"Double number is an integer";
              }
              else{
                  cout<<"Double number is not an integer";
              }
              return 0;
          }
