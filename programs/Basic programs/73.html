73. Write a C++ program to get the length of a given string 


        #include<iostream>
        using namespace std;
        
        int main() {
            char str[20];
            cout << "Enter a String : ";
            cin >> str;
            int c = 0, i = 0;
            while (str[i] != '\0') {
                c++;
                i++;
            }
            cout << endl << "Length of Entered String : " << c;
        }

Output


        Enter a String : Hello
        Length of Entered String : 5
