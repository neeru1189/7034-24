37. C++ program to find surface area of cylinder


        #include<iostream>
        using namespace std;
        
        int main() {
            float r,h;
            cout<<"Enter Radius of Cylinder : "<<endl;
            cin>>r;
            cout<<"Enter Height of Cylinder : "<<endl;
            cin>>h;
            float sa=(2*3.14*r)*(r+h);
            cout<<endl<<"Surface Area of Cylinder : "<<sa<<" square units";
        }
