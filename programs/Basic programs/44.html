44. C++ Program To Calculate CGPA Percentage


        #include<iostream>
        using namespace std;
        
        int main() {
            float marks[10],n;
            cout<<"Enter number of subjects : "<<endl;
            cin>>n;
            cout<<endl;
            float sum = 0;
            for(int i=0;i<n;i++) {
                cout<<"Enter marks of subject "<<i+1<<" : "<<endl;
                cin>>marks[i];
                sum=sum+marks[i];
            }
            float cgp=sum/(n*10);
            cout<<endl<<"CGPA : "<<cgp;
        }
