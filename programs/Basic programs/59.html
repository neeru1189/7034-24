59. Write a C++ program to display the n terms of odd natural number and their sum 


        #include<iostream>
        using namespace std;
        
        int main()
        {
            int n, sum = 0;
            cout << "Enter any number : ";
            cin >> n;
            cout << endl << "Natural odd numbers before " << n << " : ";
            for (int i = 1; i <= n; i++)
            {
                if (i % 2 != 0)
                {
                    cout << endl << i;
                    sum = sum + i;
                }
            }
            cout << endl << "Sum of odd natural numbers are " << sum;
        }
