47. C++ Program to Calculate Commission Percentage


        #include<iostream>
        using namespace std;
        
        int main()
        {
            float sales, rate;
            cout<<"Enter Total Sales : ";
            cin>>sales;
            cout<<"Enter Commission Rate : ";
            cin>>rate;
            float com = (sales * rate) / 100;
            cout<<"Total Commission : "<<com;
        }

Output


        Enter Total Sales : 1000
        Enter Commission Rate : 5
        Total Commission : 50
