39. C++ program to calculate average marks


        #include<iostream>
        using namespace std;
        
        int main() {
            float marks[10],sum=0;
            int n,i;
            cout<<"Enter number of subjects : "<<endl;
            cin>>n;
            for(i=0;i<n;i++) {
                cout<<"Enter Marks of Subject "<<i+1<<" : "<<endl;
                cin>>marks[i];
                sum=sum+marks[i];
            }
            float avg=(sum/n);
            cout<<endl<<"Total Marks : "<<sum;
            cout<<endl<<"Average Marks : "<<avg;
        }
