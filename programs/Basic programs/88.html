88. Write a C++ program to calculate HCF of Two given numbers using loop 

        
        #include<iostream>
        using namespace std;
        
        int main()
        {
            int x, y, i, c = 0;
            cout << "Enter first number : ";
            cin >> x;
            cout << "Enter second number : ";
            cin >> y;
            if (x < y)
            {
                for (i = 2; i <= x; i++)
                {
                    if (x % i == 0 && y % i == 0)
                    {
                        cout << "HCF of " << x << " and " << y << " : " << i;
                        c++;
                        break;
                    }
                }
            }
            else if (x > y)
            {
                for (i = 2; i <= y; i++)
                {
                    if (x % i == 0 && y % i == 0)
                    {
                        cout << "HCF of " << x << " and " << y << " : " << i;
                        c++;
                        break;
                    }
                }
            }
            if (c == 0)
            {
                cout << endl << "HCF of " << x << " and " << y << " : 1";
            }
        }
