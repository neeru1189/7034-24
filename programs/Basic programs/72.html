72. Write a C++ program to replace all the ‘d’ characters with ‘f’ characters 


        #include<iostream>
        #include<string.h>
        using namespace std;
        
        int main() {
            char str[20];
            cout << "Enter a string : ";
            cin >> str;
            int i = 0;
            while (str[i] != '\0') {
                if (str[i] == 'd' || str[i] == 'D') {
                    str[i] = 'f';
                }
                i++;
            }
            cout << endl << "String after altering : " << str;
        }

Output


        Enter a string : Dodge
        String after altering : fodge
