45. C++ Program to calculate compound interest


        #include<iostream>
        using namespace std;
        
        int main() {
            float p,r,t,ci;
            cout<<"Enter Principle Amount : "<<endl;
            cin>>p;
            cout<<"Enter Rate of Interest : "<<endl;
            cin>>r;
            cout<<"Enter Time : "<<endl;
            cin>>t;
            ci=(p*r*t)/100;
            cout<<endl<<"Compound Interest : "<<ci;
        }
