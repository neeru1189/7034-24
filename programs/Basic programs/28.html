28. C++ Program to find area of equilateral triangle 


        #include<iostream>
        #include<math.h>
        using namespace std;
        
        int main() {
            int a;
            cout<<"Enter side of Equilateral Triangle : "<<endl;
            cin>>a;
            float area=(sqrt(3)/4)*a*a;
            cout<<endl<<"Area of Equilateral Triangle : "<<area<<" square units";
        }
