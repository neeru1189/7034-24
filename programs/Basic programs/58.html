58. Write a C++ program to display the cube of the number upto given an integer 


        #include<iostream>
        using namespace std;
        
        int main()
        {
            int n, cube = 1;
            cout << "Enter any integer : ";
            cin >> n;
            for (int i = 1; i <= 3; i++)
            {
                cube = cube * n;
            }
            cout << endl << "Cube of " << n << " is : " << cube;
        }

Output


        Enter any integer : 2
        Cube of 2 is : 8
