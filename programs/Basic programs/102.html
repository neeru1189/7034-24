102. Write a C++ method to find number is even number or not 

        
        #include<iostream>
        using namespace std;
        int main(){
            int m;
            cout<<"Enter the number : ";
            cin>>m;
            if((m&1)==0){
                cout<<"Even number ";
            }
            else{
                cout<<"Not an even number";
            }
            return 0;
        }

Output


        Enter the number: 6
        Even number
