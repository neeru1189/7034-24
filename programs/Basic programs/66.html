66. Write a C++ program to create a simple calculator (1. ADDITION, 2. SUBTRACTION, 3. MULTIPLICATION, 4. DIVISION, 5. EXPONENTIAL, 6. PERCENTAGE, 7. EXIT) 


        #include<iostream>
        using namespace std;
        
        int main()
        {
            int x, y, ch;
            float res;
            cout << "Enter first number : ";
            cin >> x;
            cout << "Enter second number : ";
            cin >> y;
            cout << endl << "Simple Calculator : ";
            cout << endl << "1) ADDITION";
            cout << endl << "2) SUBTRACTION";
            cout << endl << "3) MULTIPLICATION";
            cout << endl << "4) DIVISION";
            cout << endl << "5) EXPONENTIAL";
            cout << endl << "6) PERCENTAGE";
            cout << endl << "7) EXIT";
            cout << endl << "Enter your choice : ";
            cin >> ch;
            if (ch == 1)
            {
                res = x + y;
                cout << endl << "Addition : " << res;
            }
            else if (ch == 2)
            {
                res = x - y;
                cout << endl << "Subtraction : " << res;
            }
            else if (ch == 3)
            {
                res = x * y;
                cout << endl << "Multiplication : " << res;
            }
            else if (ch == 4)
            {
                res = (float)x / y;
                cout << endl << "Division : " << res;
            }
            else if (ch == 5)
            {
                res = x * y;
                cout << endl << "Exponential : " << res;
            }
            else if (ch == 6)
            {
                res = (float)(x % y) / 100;
                cout << endl << "Percentage : " << res;
            }
            else if (ch == 7)
            {
                cout << endl << "Exit";
                exit(0);
            }
            else
            {
                cout << endl << "Invalid Choice!";
            }
        }
