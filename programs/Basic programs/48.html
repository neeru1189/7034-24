48. C++ Program To Find Distance Between Two Points


        #include<iostream>
        #include<math.h>
        using namespace std;
        
        int main()
        {
            int x1, x2, y1, y2;
            cout << "x and y of first point : ";
            cin >> x1 >> y1;
            cout << "x and y of second point : ";
            cin >> x2 >> y2;
            float dis = sqrt(pow(x2 - x1, 2) + pow(y2 - y1, 2));
            cout << "Distance Between first and second point : " << dis;
        }

Output


        x and y of first point : 0 0
        x and y of second point : 3 4
        Distance Between first and second point : 5
