84. Write a C++ program to find the factorial value of any number entered through the keyboard 

        
        #include<iostream>
        using namespace std;
        
        int main()
        {
            int n, fact = 1;
            cout << "Enter any number to find Factorial : ";
            cin >> n;
            for (int i = 1; i <= n; i++)
            {
                fact = fact * i;
            }
            cout << endl << "Factorial of " << n << " is : " << fact;
        }

Output

            
            Enter any number to find Factorial: 5
            
            Factorial of 5 is: 120
