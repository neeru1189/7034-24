21. Write a C++ program to compute the square root of an given integer

        
          #include<iostream>
          #include<math.h>
          using namespace std;
          int main(){
             int n;
             cout<<"Enter the number : ";
             cin>>n;
             cout<<"Square root of "<<n<<" : "<<sqrt(n);
             return 0;
          }

Output

            
              Enter the number : 25
              Square root of 25 : 5
