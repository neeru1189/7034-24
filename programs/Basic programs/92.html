92. Write a C++ program to count total number of notes in entered amount using loop 

        
        #include<iostream>
        #include<string.h>
        using namespace std;
        
        int main()
        {
            char num[20];
            int c = 0;
            cout << "Enter amount : ";
            cin >> num;
            int l = strlen(num);
            for (int i = 0; i < l; i++)
            {
                if (num[i] == ',')
                {
                    c++;
                }
            }
            cout << endl << "Number of notes in entered amount " << num << " : " << c;
        }

Output

            
            Enter amount: 1,000,000
            
            Number of notes in entered amount 1,000,000 : 2
