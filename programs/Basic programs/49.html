49. C++ Program To Calculate Power Of Number


        #include<iostream>
        using namespace std;
        
        int main()
        {
            int n, p, ans = 1;
            cout << "Enter number and its power : ";
            cin >> n >> p;
            for (int i = 1; i <= p; i++)
            {
                ans = ans * n;
            }
            cout << endl << n << " raise to power " << p << " = " << ans;
        }

Output


        Enter number and its power : 2 3
