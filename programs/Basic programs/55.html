55. Write a C++ program to input 5 numbers from keyboard and find their sum and average 


        #include<iostream>
        using namespace std;
        
        int main()
        {
            int n[5], sum = 0;
            float avg;
            for (int i = 0; i < 5; i++)
            {
                cout << "Enter number " << i + 1 << " : ";
                cin >> n[i];
                sum = sum + n[i];
            }
            avg = (float)sum / 5;
            cout << endl << "Sum of entered numbers is : " << sum;
            cout << endl << "Average of entered numbers is : " << avg;
        }
