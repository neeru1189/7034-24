93. Write a C++ program to print Fibonacci series of n terms where n is input by user using loop 

        
        #include<iostream>
        using namespace std;
        
        int main()
        {
            int n, t1 = 0, t2 = 1, nt;
            cout << "Enter number of terms in Fibonacci Series : ";
            cin >> n;
            cout << endl << "Fibonacci Series : " << endl;
            if (n == 0)
            {
                cout << "Zero terms printed";
            }
            else if (n == 1)
            {
                cout << "\t0";
            }
            else if (n == 2)
            {
                cout << "\t0\t1";
            }
            else
            {
                cout << "\t0\t1";
                int c = 3;
                while (c <= n)
                {
                    nt = t1 + t2;
                    t1 = t2;
                    t2 = nt;
                    cout << "\t" << nt;
                    c++;
                }
            }
        }
