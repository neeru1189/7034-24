35. C++ Program to find volume of cone


        #include<iostream>
        using namespace std;
        
        int main() {
            int h,r;
            cout<<"Enter Height of Cone : "<<endl;
            cin>>h;
            cout<<"Enter Radius of Cone : "<<endl;
            cin>>r;
            float volume=(h*3.14*r*r)/3;
            cout<<endl<<"Volume of Cone : "<<volume<<" cubic units";
        }
