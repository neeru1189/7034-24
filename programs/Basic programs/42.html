42. C++ program to find factorial of any number


        #include<iostream>
        using namespace std;
        
        int main() {
            int n;
            cout<<"Enter Any number to find factorial : "<<endl;
            cin>>n;
            int fact=1;
            for(int i=1;i<=n;i++) {
                fact=fact*i;
            }
            cout<<endl<<"Factorial of "<<n<<" is : "<<fact;
        }

Output


        Enter Any number to find factorial : 
        6
        Factorial of 6 is : 720
