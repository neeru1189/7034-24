4. Write a C++ program to print the sum, multiply, subtract, divide and remainder of two numbers

        
          #include<iostream>
          using namespace std;
          int main(){
              int a,b;
              cout<<"Enter number A : ";
              cin>>a;
              cout<<"Enter number B : ";
              cin>>b;
              cout<<"Sum of A and B : "<<a+b;
              cout<<endl<<"Product of A and B : "<<a*b;
              cout<<endl<<"Subtract of A from B : "<<b-a;
              cout<<endl<<"Divison of A by B : "<<a/b;
              cout<<endl<<"Remainder of A by B : "<<a%b;
              return 0;
          }

Output

        
          Enter number A : 10
          Enter number B : 5
          Sum of A and B : 15
          Product of A and B : 50
          Subtract of A from B : -5
          Divison of A by B : 2
          Remainder of A by B : 0
