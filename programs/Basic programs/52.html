52. Write a C++ program to test a number is positive or negative


        #include<iostream>
        using namespace std;
        
        int main()
        {
            int n;
            cout << "Enter any number : ";
            cin >> n;
            if (n >= 0)
            {
                cout << "Entered number is positive";
            }
            else
            {
                cout << "Entered number is negative";
            }
        }
