89. Write a C++ program to enter the numbers till the user wants and at the end it should display the count of positive, negative and zeros 

        
        #include<iostream>
        using namespace std;
        
        int main()
        {
            int n[20], x;
            int cp = 0, cn = 0, cz = 0, i;
            cout << "Enter total numbers of integers : ";
            cin >> x;
            for (i = 0; i < x; i++)
            {
                cout << "Enter number " << i + 1 << " : ";
                cin >> n[i];
        
                if (n[i] > 0)
                {
                    cp++;
                }
                else if (n[i] < 0)
                {
                    cn++;
                }
                else
                {
                    cz++;
                }
            }
            cout << endl << "Count of positive numbers : " << cp;
            cout << endl << "Count of negative numbers : " << cn;
            cout << endl << "Count of zeros : " << cz;
        }
