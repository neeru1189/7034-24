87. Write a C++ program to check whether the number is a prime number or not 

        
        #include<iostream>
        using namespace std;
        
        int main()
        {
            int n, c = 0;
            cout << "Enter any number : ";
            cin >> n;
            for (int i = 2; i <= (n / 2); i++)
            {
                if (n % i == 0)
                {
                    c++;
                    break;
                }
            }
            if (c == 0)
            {
                cout << endl << n << " is a prime number";
            }
            else
            {
                cout << endl << n << " is not a prime number";
            }
        }
