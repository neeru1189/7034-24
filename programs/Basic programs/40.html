40. C++ program to check vowel or consonant


        #include<iostream>
        using namespace std;
        
        int main() {
            char c;
            cout<<"Enter a character : "<<endl;
            cin>>c;
            switch(c) {
                case 'a':
                case 'A':
                case 'e':
                case 'E':
                case 'i':
                case 'I':
                case 'o':
                case 'O':
                case 'u':
                case 'U':
                    cout<<"Entered Character is a vowel";
                    break;
                default:
                    cout<<"Entered Character is a consonant";
            }
        }

Output

        Enter a character : 
        A
        Entered Character is a vowel
