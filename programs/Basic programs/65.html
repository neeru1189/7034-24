        #include<iostream>
        using namespace std;
        
        int main()
        {
            int marks[10], n, sum = 0;
            float avg;
            cout << "Enter number of subjects (1-10): ";
            cin >> n;
            for (int i = 0; i < n; i++)
            {
                cout << "Enter marks of subject " << i + 1 << " : ";
                cin >> marks[i];
                sum = sum + marks[i];
            }
            avg = (float)sum / n;
            cout << endl << "Sum of marks obtained by student : " << sum;
            cout << endl << "Average of student : " << avg;
            if (avg > 81)
            {
                cout << endl << "Grade A";
            }
            else if (avg < 81 && avg > 61)
            {
                cout << endl << "Grade B";
            }
            else if (avg < 61 && avg > 41)
            {
                cout << endl << "Grade C";
            }
            else
            {
                cout << endl << "Grade D";
            }
        }
