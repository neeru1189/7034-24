2. Write a C++ program to create an abstract class Animal with an abstract method called sound(). Create derived classes Lion and Tiger that extend the Animal class and implement the sound() method to make a specific sound for each animal.

        
          #include <iostream>
      
          class Animal {
          public:
              
              virtual void sound() const = 0;
          };
      
          class Lion : public Animal {
          public:
             
              void sound() const override {
                  std::cout << "Lion roars" << std::endl;
              }
          };
      
          class Tiger : public Animal {
          public:
           
              void sound() const override {
                  std::cout << "Tiger growls" << std::endl;
              }
          };
      
          int main() {
           
              Lion lion;
              Tiger tiger;
      
             
              std::cout << "Sound of Lion: ";
              lion.sound();
      
              std::cout << "Sound of Tiger: ";
              tiger.sound();
      
              return 0;
          }

Output

        
          Sound of Lion: Lion roars
          Sound of Tiger: Tiger growls
