11. Write a C++ program to create an abstract class Shape2D with abstract methods draw() and resize(). Create derived classes Rectangle and Circle that extend the Shape2D class and implement the respective methods to draw and resize each shape. 

        
        #include <iostream>
        
        class Shape2D {
        public:
            virtual void draw() const = 0;
            virtual void resize(double factor) = 0;
        };
        
        class Rectangle : public Shape2D {
        private:
            double width;
            double height;
        
        public:
            Rectangle(double w, double h) : width(w), height(h) {}
        
            void draw() const override {
                std::cout << "Drawing Rectangle with width: " << width << " and height: " << height << std::endl;
            }
        
            void resize(double factor) override {
                width *= factor;
                height *= factor;
                std::cout << "Resizing Rectangle by factor: " << factor << std::endl;
            }
        };
        
        class Circle : public Shape2D {
        private:
            double radius;
        
        public:
            Circle(double r) : radius(r) {}
        
            void draw() const override {
                std::cout << "Drawing Circle with radius: " << radius << std::endl;
            }
        
            void resize(double factor) override {
                radius *= factor;
                std::cout << "Resizing Circle by factor: " << factor << std::endl;
            }
        };
        
        int main() {
            Rectangle rectangle(5, 3);
            std::cout << "Rectangle:" << std::endl;
            rectangle.draw();
            rectangle.resize(1.5);
            rectangle.draw();
        
            std::cout << std::endl;
        
            Circle circle(4);
            std::cout << "Circle:" << std::endl;
            circle.draw();
            circle.resize(2);
            circle.draw();
        
            return 0;
        }
