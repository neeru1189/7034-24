12. Write a C++ program to create a class called "Airplane" with a flight number, destination, and departure time attributes, and methods to check flight status and delay. 


        #include <iostream>
        #include <string>
        #include <ctime>
        
        class Airplane {
        private:
            std::string a;
            std::string b;
            time_t c;
        
        public:
            Airplane(const std::string& a, const std::string& b, time_t c) : a(a), b(b), c(c) {}
        
            void d() const {
                time_t currentTime = time(nullptr);
                if (currentTime > c) {
                    std::cout << "Delayed" << std::endl;
                } else {
                    std::cout << "On time" << std::endl;
                }
            }
        
            void disp() const {
                std::cout << "Flight " << a << " to " << b << ", Departure time: " << c << ", Status: ";
                d();
            }
        };
        
        int main() {
            Airplane airplane("ABC123", "New York", time(nullptr) + 3600);
        
            airplane.disp();
        
            return 0;
        }

Output


            Flight ABC123 to New York, Departure time: [Current time + 3600], Status: On time
