        #include <iostream>
        #include <string>
        #include <iomanip> // For std::setprecision
        
        class Employee {
        private:
            int employee_id;
            std::string employee_name;
            double employee_salary;
        
        public:
            void setEmployeeID(int id) {
                employee_id = id;
            }
        
            int getEmployeeID() const {
                return employee_id;
            }
        
            void setEmployeeName(const std::string& name) {
                employee_name = name;
            }
        
            std::string getEmployeeName() const {
                return employee_name;
            }
        
            void setEmployeeSalary(double salary) {
                employee_salary = salary;
            }
        
            std::string getFormattedSalary() const {
                std::stringstream ss;
                ss << std::fixed << std::setprecision(2) << employee_salary;
                return "$" + ss.str();
            }
        };
        
        int main() {
            Employee emp;
        
            emp.setEmployeeID(101);
            emp.setEmployeeName(" sachin");
            emp.setEmployeeSalary(500);
        
            std::cout << "Employee ID: " << emp.getEmployeeID() << std::endl;
            std::cout << "Employee Name: " << emp.getEmployeeName() << std::endl;
            std::cout << "Employee Salary: " << emp.getFormattedSalary() << std::endl;
        
            return 0;
        }

Output

            
            Employee ID: 101
            Employee Name:  sachin
            Employee Salary: $500.00
