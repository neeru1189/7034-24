4. Program to subtract the two matrices 

        
          #include<iostream>
          #include<iomanip>
          using namespace std;
      
          int main() {
              int x, y, i, j;
              cout << "Enter the rows and columns for the matrices: ";
              cin >> x >> y;
      
              int matrixA[10][10], matrixB[10][10], result[10][10];
      
              cout << "Enter elements of Matrix A: " << endl;
              for (i = 0; i < x; i++) {
                  for (j = 0; j < y; j++) {
                      cin >> matrixA[i][j];
                  }
              }
      
              cout << "Enter elements of Matrix B: " << endl;
              for (i = 0; i < x; i++) {
                  for (j = 0; j < y; j++) {
                      cin >> matrixB[i][j];
                  }
              }
      
              // Subtract matrices
              for (i = 0; i < x; i++) {
                  for (j = 0; j < y; j++) {
                      result[i][j] = matrixA[i][j] - matrixB[i][j];
                  }
                  class="language-cpp"
              }
      
              cout << "Resultant Matrix: " << endl;
              for (i = 0; i < x; i++) {
                  for (j = 0; j < y; j++) {
                      cout << setw(5) << result[i][j];
                  }
                  cout << endl;
                }
                
              return 0;
          }
