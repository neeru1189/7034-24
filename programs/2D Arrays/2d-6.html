6. Program to display the lower triangular matrix 

         
          #include<iostream>
          using namespace std;
          int main(){
              int x, y, i, j, c=0;
              cout<<"Enter the rows and columns For Matrix: ";
              cin>>x>>y;
              if(x==y){
                  int mata[10][10];
                  cout<<"Enter the elements of matrix : "<<endl;
                  for(i=0;i<x;i++){
                      for(j=0;j<y;j++){
                          cin>>mata[i][j];
                      }
                  }
                  cout<<"Lower triangular matrix of the given matrix : "<<endl;
                  for(i=0;i<x;i++){
                      for(j=0;j<y;j++){
                          if (i==j){
                              c=1;
                              cout<<"  "<<mata[i][j];
                          }
                          else if (c==0){
                              cout<<"  "<<mata[i][j];
                          }
                          else{
                              cout<<"  "<<0;
                          }
                      }
                      cout<<endl;
                      c=0;
                  }
              }
              else{
                  cout<<"Not a square matrix";
              }
              return 0;
          }
