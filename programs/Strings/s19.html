19. Program to find Reverse of the string 

        
        #include <iostream>
        #include <string>
        
        using namespace std;
        
        int main() {
            string str;
            cout << "Enter a string: ";
            cin >> str;
        
            int n = str.length();
        
            for (int i = 0; i < n / 2; i++) {
                char temp = str[i];
                str[i] = str[n - i - 1];
                str[n - i - 1] = temp;
            }
            cout << "Reverse of the string: " << str << endl;
            return 0;
        }

Output

            
            Enter a string: hello
            Reverse of the string: olleh
